Useful built-in decorators, first half

class Rectangle(object):
  def __init__(self, width, height):
    self._width = self.CheckPositiveDimension(width)  
    self._height = self.CheckPositiveDimension(height)

  @staticmethod
  def CheckPositiveDimension(x):
    x = float(x)
    assert x > 0
    return x

  @property
  def width(self): return self._width

  @property
  def height(self): return self._height

print Rectangle.CheckPositiveDimension(42)  #: 42.0
print Rectangle(8, 9).width  #: 8

Useful built-in decorators, second half

class LandscapeRectangle(Rectangle):
  def __init__(self, width, height):
    self.CheckLandscape(width, height)
    Rectangle.__init__(self, width, height)

  @classmethod
  def CheckLandscape(cls, width, height):
    width = cls.CheckPositiveDimension(width)
    height = cls.CheckPositiveDimension(height)
    assert width >= height

LandscapeRectangle.CheckLandscape(6, 7)  # AssertionError
  • @property creates a so-called descriptor (see more here).
  • def creates a function, which class wraps into an instancemethod object. No other wrapping occurs.
  • @staticmethod creates a staticmethod object. No wrapping occurs. Likewise for @classmethod.

Solution which checks at runtime

def rt_pure(f):
  builtins = type(__import__('sys'))('restricted_builtins')
  builtins.__dict__.update(f.func_globals['__builtins__'].__dict__)
  builtins.__dict__.pop('__import__', None)
  return type(f)(f.func_code, {'__builtins__': builtins},
                 f.func_name, f.func_defaults, f.func_closure)

@rt_pure
def median(xs):
  import sys  # ImportError: __import__ not found
  x = answer  # NameError: global name 'answer' is not defined
  print xs  # Problem #1: Works, but it shouldn't.
  #global answer; answer += 1  # Would work, but no global effect.
  return (xs[len(xs) / 2] + xs[(len(xs) - 1) / 2]) / 2

# Problem #2: Does not restrict inner functions.
# Problem #3 (runtime): Problem not detected until call.

Solution which checks at compile time

import opcode

def pure(f):
  co = f.func_code
  for name in co.co_names:
    assert name not in ('__import__', 'globals')
    assert hasattr(__builtins__, name)
  code = co.co_code
  i = 0
  while i < len(code):
    op = ord(code[i])
    i += 1
    if op >= opcode.HAVE_ARGUMENT: i += 2
    assert opcode.opname[op] not in (
        'STORE_GLOBAL', 'DELETE_GLOBAL', 'PRINT_EXPR',
        'PRINT_ITEM', 'IMPORT_NAME', 'IMPORT_FROM')
  return f

Problem: Relieas on CPython implementation details, may have to be adjusted between major versions (2.6 to 2.7 to 3.3) etc.

What is a restriction?

Restriction: a runtime or compile-time mechanism to prevent unintended use.

  • assertion
  • abstract class
  • abstract method
  • final method
  • override method
  • pure function
  • finalim class method
  • nosuper method

Java has abstract, final, @Override. C# has abstract, sealed and override.

Definition of various restrictions

  • abstract class: It is an error to instantiate an abstract class.
  • abstract method: It is an error to instantiate a class with unoverridden abstract methods.
  • final method: It is an error to override a final method in a descendant class.
  • override method: It is an error to create a class with an override method, without an ancestor class having a method with the same signature (name).
  • pure function: It is an error to create a pure function which reads or writes global (or enclosing-scope) variables.
  • finalim class method: It is an error to override a finalim method with an instance method in a descendant class.
  • nosuper method: It is error to create a class with a nosuper method, with an ancestor class having a mehtod with the same signature (name).

Options for implementing restrictions

  • Add them as comments, and make the lint tool verify them.
  • Add them as no-op syntax elements (e.g. using decorators), and make the lint tool verify them.
  • If a compile-time check is not feasible, then add them as syntax elements and make them fail at runtime.
  • Add them as syntax elements (e.g. using decorators) and make the compiler verify them.

(The later the option, the better.)

@abstract_class with class decorators

def abstract_class(k):
  def abstract_init(cls, *args, **kwargs):
    if cls is k:
      raise RuntimeError('Cannot instantiate abstract %r.' % cls)
  k.__init__ = classmethod(abstract_init)
  return k

@abstract_class  # Needs Python 2.6 or later.
class Foo(object):
  answer = 42

class Bar(Foo):
  pass

print Foo.answer
print Bar()  # @abstract_class is not inherited.
print Foo('foo')  # RuntimeError: Cannot instantiate...

This approach won't work for @abstract or @final, because they affect descendant classes at compile time.

Expected behavior of @abstract

@abstract makes the class and descendant object creation fail if the class contains an abstract method. 

class Foo(object):
  answer = 42
  @abstract
  def eat(self): pass  # No implementation.

class Bar(Foo):
  pass

class Baz(Bar):
  def eat(self): print 'EATING'

print Bar.answer  #: 42
print Baz()
print type(type(Baz()))  #: <class '__main__.CheckerMeta'>
print Bar()  # RuntimeError: Cannot instantiate...
print Foo()  # RuntimeError: Cannot instantiate...

Implementation of @abstract, first half

import sys
import types

def abstract(f):
  f.is_abstract = True
  fb = sys._getframe().f_back
  assert '__module__' in fb.f_locals
  metaclass = fb.f_locals.get('__metaclass__') or type # new-style
  assert metaclass in (type, CheckerMeta)  # TODO: generic
  fb.f_locals['__metaclass__'] = CheckerMeta
  return f
  • @abstract sets a flag (is_abstract) and injects a metaclass.
  • The metaclass gets run when that class or a descendant class of it gets created, and will do the appropriate checks to implement the restriction.

Implementation of @abstract, second half

class CheckerMeta(type):
  def __new__(cls, name, bases, dict_obj):
    newcls = type.__new__(cls, name, bases, dict_obj)
    has_abstract = False
    for key in dir(newcls):
      value = getattr(newcls, key)
      if isinstance(value, types.MethodType):
        if getattr(value, 'is_abstract', None):
          has_abstract = True
          break
    if has_abstract:
      def abstract_init(acls, *args, **kwargs):
        if acls is newcls:
          raise RuntimeError(
              'Cannot instantiate abstract %r.' % acls)
      newcls.__init__ = classmethod(abstract_init)
    return newcls

Expected behavior of @final

@final makes the descendant class definition (not object creation!) fail if it contains a method with the same name. 

class Foo:  # Will be upgraded to new style.
  answer = 42
  @abstract
  def eat(self): pass  # No implementation.

class Bar(Foo):
  @final
  def drink(self): print 'DRINKING'

class Baz(Bar):
  def eat(self): print 'EATING'

# RuntimeError: Cannot override final methods: drink.
class Quux(Baz):
  def drink(self): print 'BAD DRINK'

Implementation of @final, first half

import sys
import types

def install_checker_meta(fb):
  assert '__module__' in fb.f_locals
  metaclass = fb.f_locals.get('__metaclass__') or type # new-style
  assert metaclass in (type, CheckerMeta)  # TODO: generic
  fb.f_locals['__metaclass__'] = CheckerMeta
def abstract(f):
  assert not getattr(f, 'is_final', None)
  f.is_abstract = True
  install_checker_meta(sys._getframe().f_back)
  return f
def final(f):
  assert not getattr(f, 'is_abstract', None)
  f.is_final = True
  install_checker_meta(sys._getframe().f_back)
  return f

Implementation of @final, second half

class CheckerMeta(type):
  def __new__(cls, name, bases, dict_obj):
    final_methods = set(key for base in bases for key in dir(base)
        if isinstance(getattr(base, key), types.MethodType) and
        getattr(getattr(base, key), 'is_final', None))
    final_methods.intersection_update(dict_obj)
    if final_methods:
      raise RuntimeError('Cannot override final methods: %s.' %
                         ', '.join(sorted(final_methods)))
    newcls = type.__new__(cls, name, bases, dict_obj)
    abstract_methods = set(key for key in dir(newcls) if
        isinstance(getattr(newcls, key), types.MethodType) and
        getattr(getattr(newcls, key), 'is_abstract', None))
    if abstract_methods:
      def abstract_init(acls, *args, **kwargs):
        if acls is newcls:
           raise RuntimeError('Cannot instantiate @abstract: %s.'
               % ', '.join(sorted(abstract_methods)))
      newcls.__init__ = classmethod(abstract_init)
    return newcls

Usage

from pobjects import *

class Foo(object):
  answer = 42
  @abstract  # Comes from the module pobjects.
  def eat(self): pass

# pobject makes itself work without imports in each .py file:
__builtins__['final'] = final
__builtins__['abstract'] = abstract
  • pobjects, the reference implementation has been released under the GPL.
  • pobjects supports adding checks to classes with an existing metaclass.
  • pobjects reports detailed error messages, adding all of them to an exception.
